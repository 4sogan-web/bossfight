  <script type="text/babel">

    const {useEffect,useMemo,useRef,useState} = React;


    // ---- Tema objesi (JS tarafı)

    const COLORS = {

      bg:"var(--bg)", teal:"var(--teal)", mint:"var(--mint)",

      card:"var(--card)", text:"var(--text)", dim:"var(--dim)", border:"var(--border)"

    };


    // ---- Yardımcılar

    const todayKey = () => {

      const d = new Date(), y=d.getFullYear(), m=String(d.getMonth()+1).padStart(2,"0"), day=String(d.getDate()).padStart(2,"0");

      return `${y}${m}${day}`;

    };

    const getDailyCount = () => {

      const key=`bfgDaily_${todayKey()}`; const raw=localStorage.getItem(key);

      return { key, count: raw?Number(raw):0 };

    };

    const incDailyCount = () => { const {key,count}=getDailyCount(); const next=count+1; localStorage.setItem(key,String(next)); return next; };

    const sentenceSplit = t => (t||"").split(/(?<=[.!?])\s+/).filter(Boolean);

    const naiveSummaryBullets = (t,max=3) => {

      const s=sentenceSplit(t); const b=s.slice(0,max);

      while(b.length<max) b.push("(Özet için içerik yetersiz)");

      return b;

    };

    const SAMPLE_NEWS = [

      { title:"Deepfake düzenlemesi Meclis'te görüşülüyor",

        source:"Anadolu Ajansı",

        content:"Meclis'te dijital manipülasyon ve deepfake içeriklerine yönelik yeni düzenleme taslağı ele alınıyor. Taslak, seçim dönemlerinde yapay içeriklerin açık şekilde etiketlenmesini öngörüyor. Platformlara, şeffaflık raporu ve hızlı kaldırma yükümlülüğü getiriliyor." },

      { title:"Dijital telif platformu içerik üreticilerine doğrudan ödeme başlattı",

        source:"Bloomberg HT",

        content:"Türkiye merkezli bir girişim, haber ve makale alıntıları için içerik üreticilerine telif dağıtan aracı bir katman duyurdu. Yayıncılar ve platformlar arasında şeffaf bir API ile tıklama ve alıntı bazlı dağıtım yapılacak. İlk anlaşmalar imzalanmaya başladı." },

    ];


    // ---- Puanlama

    const SCORING = {

      q1:{ correct:100, wrong:-20 },

      q2:{ correct:150 },

      sliderMax:120,

      prediction:{ provisional:80 },

      memePerChar:1, memeMax:60,

      streak:[1,1.05,1.1,1.15,1.2,1.22,1.25],

    };

    const sliderScore = (guess,target=68) => Math.round(Math.max(0, SCORING.sliderMax - Math.abs(guess-target)*2));

    const memeScore = t => Math.min(SCORING.memeMax, (t||"").trim().length * SCORING.memePerChar);

    const streakMultiplierFromLocal = () => {

      const raw=localStorage.getItem("bfgStreak");

      const last=localStorage.getItem("bfgStreakLast");

      const today=todayKey(); let streak=raw?Number(raw):0;

      if(last && last!==today){

        const dLast=new Date(`${last.slice(0,4)}-${last.slice(4,6)}-${last.slice(6,8)}`);

        const diff=Math.floor((new Date()-dLast)/(1000*60*60*24));

        if(diff===1) streak=Math.min(streak+1,7); else if(diff>1) streak=1;

      } else if(!last){ streak=1; }

      localStorage.setItem("bfgStreak",String(streak));

      localStorage.setItem("bfgStreakLast",today);

      return SCORING.streak[Math.min(streak-1, SCORING.streak.length-1)];

    };


    // ---- SVG → PNG

    const downloadKillCardPNG = async (svgRef, fileName="killcard.png") => {

      if(!svgRef.current) return;

      const svgEl=svgRef.current;

      const serializer=new XMLSerializer();

      const src=serializer.serializeToString(svgEl);

      const svg64=btoa(unescape(encodeURIComponent(src)));

      const img=new Image();

      const w=svgEl.viewBox.baseVal.width||1080, h=svgEl.viewBox.baseVal.height||1920, scale=2;

      return new Promise(resolve=>{

        img.onload=()=>{

          const canvas=document.createElement("canvas");

          canvas.width=w*scale; canvas.height=h*scale;

          const ctx=canvas.getContext("2d");

          ctx.fillStyle=getComputedStyle(document.documentElement).getPropertyValue('--bg').trim();

          ctx.fillRect(0,0,canvas.width,canvas.height);

          ctx.drawImage(img,0,0,canvas.width,canvas.height);

          const a=document.createElement("a"); a.href=canvas.toDataURL("image/png"); a.download=fileName; a.click();

          resolve(true);

        };

        img.src="data:image/svg+xml;base64,"+svg64;

      });

    };


    // ---- UI parçaları

    const Chip = ({children}) => (

      <span className="inline-flex items-center px-2 py-1 text-xs rounded-full"

            style={{background:"#0f1a20", color:COLORS.mint, border:`1px solid ${COLORS.border}`}}>

        {children}

      </span>

    );

    const Card = ({children}) => (

      <div className="rounded-2xl p-5 shadow-lg" style={{background:COLORS.card, border:`1px solid ${COLORS.border}`}}>

        {children}

      </div>

    );

    const Button = ({children,onClick,variant="primary",disabled}) => {

      const base="px-4 py-2 rounded-xl text-sm font-semibold transition-opacity";

      const style = variant==="outline"

        ? { color:COLORS.teal, border:`1px solid ${COLORS.teal}`, background:"transparent" }

        : { background:COLORS.teal, color:"#07241F" };

      return <button disabled={disabled} onClick={onClick}

                     className={`${base} ${disabled?"opacity-50":"hover:opacity-90"}`} style={style}>{children}</button>;

    };

    const SectionTitle = ({children}) => <h2 className="text-xl font-bold mb-3" style={{color:COLORS.text}}>{children}</h2>;

    const Text = ({children,dim}) => <p className={`text-sm ${dim?"opacity-80":""}`} style={{color:COLORS.text}}>{children}</p>;


    // ---- Kill Card SVG

    const KillCardSVG = ({ title, source, score, streak, bullets, svgRef }) => {

      const w=1080, h=1920;

      return (

        <svg ref={svgRef} viewBox={`0 0 ${w} ${h}`} width={w} height={h} xmlns="http://www.w3.org/2000/svg">

          <defs>

            <linearGradient id="g" x1="0" x2="1" y1="0" y2="1">

              <stop offset="0%" stopColor="#0A1217"/><stop offset="100%" stopColor="#0E1E22"/>

            </linearGradient>

          </defs>

          <rect x="0" y="0" width={w} height={h} fill="url(#g)" />

          <rect x="60" y="80" width={w-120} height="170" rx="24" fill="#0f1a20" stroke="#2A3A42" />

          <text x="100" y="160" fill="#E9F0F2" fontFamily="Inter, system-ui" fontWeight="700" fontSize="48">BossFight Gündem</text>

          <text x="100" y="210" fill="#2BD9AF" fontFamily="Inter, system-ui" fontSize="28">Kill Card</text>


          <rect x="60" y="280" width={w-120} height="500" rx="24" fill="#121A20" stroke="#2A3A42" />

          <text x="100" y="350" fill="#E9F0F2" fontFamily="Inter, system-ui" fontWeight="700" fontSize="40">{title}</text>

          <text x="100" y="400" fill="#AAB7BF" fontFamily="Inter, system-ui" fontSize="28">Kaynak: {source}</text>

          {bullets.slice(0,3).map((b,i)=>(

            <text key={i} x="100" y={460+i*60} fill="#E9F0F2" fontFamily="Inter, system-ui" fontSize="28">• {b.length>80?b.slice(0,77)+"…":b}</text>

          ))}


          <rect x="60" y="820" width={w-120} height="400" rx="24" fill="#121A20" stroke="#2A3A42" />

          <text x="100" y="890" fill="#2BD9AF" fontFamily="Inter, system-ui" fontWeight="700" fontSize="36">Skor</text>

          <text x="100" y="950" fill="#E9F0F2" fontFamily="Inter, system-ui" fontWeight="800" fontSize="80">{score}</text>

          <text x="100" y="1020" fill="#AAB7BF" fontFamily="Inter, system-ui" fontSize="26">Streak: x{streak.toFixed(2)}</text>

          <text x="100" y="1080" fill="#AAB7BF" fontFamily="Inter, system-ui" fontSize="24">Doğruluk, kaynak ve slider puanları baz alınmıştır.</text>


          <rect x="60" y="1260" width={w-120} height="520" rx="24" fill="#121A20" stroke="#2A3A42" />

          <text x="100" y="1330" fill="#2BD9AF" fontFamily="Inter, system-ui" fontWeight="700" fontSize="36">Rozetler</text>

          <text x="100" y="1400" fill="#E9F0F2" fontFamily="Inter, system-ui" fontSize="28">🏅 Simülakr Avcısı • 🧭 Kaynak Avcısı • 🧠 Anima Analisti</text>

          <text x="100" y="1480" fill="#AAB7BF" fontFamily="Inter, system-ui" fontSize="24">bossfightgundem.com</text>

        </svg>

      );

    };


    // ---- Uygulama

    function App(){

      const [step,setStep]=useState("landing");

      const [limitReached,setLimitReached]=useState(false);

      const [news,setNews]=useState(SAMPLE_NEWS[0]);

      const [newsText,setNewsText]=useState("");

      const [bullets,setBullets]=useState([]);

      const [counterView,setCounterView]=useState(0);

      const [q1,setQ1]=useState(null), [q2,setQ2]=useState(null), [q3,setQ3]=useState(50), [q4,setQ4]=useState(null), [q5,setQ5]=useState("");

      const [score,setScore]=useState(0), [streakMult,setStreakMult]=useState(1);

      const svgRef=useRef(null);


      useEffect(()=>{ const {count}=getDailyCount(); setCounterView(count); setLimitReached(count>=3); },[step]);


      const prepareSummary=(n)=>{

        const txt=newsText.trim()||n.content;

        const b=naiveSummaryBullets(txt,3);

        const oppose="Karşıt görüş: Aynı veriler farklı bağlamda farklı sonuçlar doğurabilir.";

        setBullets([...b, oppose]);

      };


      const startFlow=(n)=>{

        const {count}=getDailyCount();

        if(count>=3){ setLimitReached(true); return; }

        const next=incDailyCount();

        setCounterView(next);

        setNews(n);

        prepareSummary(n);

        setStreakMult(streakMultiplierFromLocal());

        setQ1(null); setQ2(null); setQ3(50); setQ4(null); setQ5(""); setScore(0);

        setStep("summary");

      };


      const computeScore=()=>{

        let s=0;

        if(q1===true) s+=100; else if(q1===false) s+=-20;

        if(q2===0) s+=150;

        s+=sliderScore(q3,68);

        if(q4!==null) s+=80;

        s+=memeScore(q5);

        s=Math.round(s*streakMult);

        setScore(s);

      };


      useEffect(()=>{ if(step==="killcard") computeScore(); },[step]);


      const options = useMemo(()=>[news.source,"Kaynak A","Kaynak B"],[news.source]);


      const Landing=()=>(

        <div className="max-w-5xl mx-auto p-6">

          <header className="flex items-center justify-between mb-6">

            <h1 className="text-2xl font-extrabold" style={{color:COLORS.text}}>BossFight Gündem</h1>

            <div className="flex items-center gap-2">

              <Chip>Günlük limit: 3 haber</Chip>

              <Chip>Bugün: {counterView}/3</Chip>

            </div>

          </header>


          {limitReached && (

            <Card><p className="text-sm" style={{color:COLORS.text}}>Günlük haber limitine ulaştın. Yarın tekrar gel; streak çarpanın korunur.</p></Card>

          )}


          <div className="grid md:grid-cols-3 gap-4 mt-4">

            {SAMPLE_NEWS.map((n,idx)=>(

              <Card key={idx}>

                <div className="flex flex-col gap-2">

                  <Chip>{n.source}</Chip>

                  <h3 className="text-lg font-bold" style={{color:COLORS.text}}>{n.title}</h3>

                  <Text dim>{n.content.slice(0,120)}…</Text>

                  <div className="flex gap-2 mt-2">

                    <Button disabled={limitReached} onClick={()=>startFlow(n)}>Bu haberi oyna</Button>

                  </div>

                </div>

              </Card>

            ))}

          </div>


          <div className="mt-6 grid md:grid-cols-2 gap-4">

            <Card>

              <SectionTitle>URL ya da metinle başla</SectionTitle>

              <Text dim>Metni kutuya yapıştır; özet ve oyun akışı hemen oluşur.</Text>

              <textarea className="w-full mt-3 bg-transparent rounded-xl p-3 border"

                        style={{borderColor:COLORS.border, color:COLORS.text}} rows="5"

                        placeholder="Haber metnini buraya yapıştır (demo: 3-4 cümle yeterli)"

                        value={newsText} onChange={(e)=>setNewsText(e.target.value)} />

              <div className="mt-3 flex gap-2">

                <Button disabled={limitReached} onClick={()=>startFlow({

                  title:"Kendi metninle dene", source:"Kullanıcı",

                  content: newsText || "Kısa bir metin örneği. Bu cümle bir özet oluşturur. İkinci cümle akışı tamamlar. Üçüncü cümle de bağlam ekler."

                })}>Metnimle başla</Button>

                <Button variant="outline" onClick={()=>{localStorage.clear(); location.reload();}}>Limit & streak sıfırla</Button>

              </div>

            </Card>

          </div>

        </div>

      );


      const Summary=()=>(

        <div className="max-w-3xl mx-auto p-6">

          <div className="flex items-center justify-between mb-4">

            <Chip>{news.source}</Chip>

            <Chip>Streak x{streakMult.toFixed(2)}</Chip>

          </div>

          <Card>

            <h2 className="text-2xl font-bold mb-2" style={{color:COLORS.text}}>{news.title}</h2>

            <SectionTitle>Turbo Özet (60 sn)</SectionTitle>

            <ul className="list-disc pl-6" style={{color:COLORS.text}}>

              {bullets.map((b,i)=>(<li key={i} className="mb-1">{b}</li>))}

            </ul>

            <div className="mt-4 flex gap-2">

              <Button onClick={()=>setStep("quiz")}>Mini Oyuna Geç</Button>

              <Button variant="outline" onClick={()=>setStep("landing")}>Geri</Button>

            </div>

          </Card>

        </div>

      );


      const Quiz=()=>(

        <div className="max-w-3xl mx-auto p-6 space-y-4">

          <Card>

            <SectionTitle>A1: İddia Doğru mu?</SectionTitle>

            <Text dim>{news.title}</Text>

            <div className="mt-3 flex gap-2">

              <Button variant={q1===true?"":"outline"} onClick={()=>setQ1(true)}>Doğru</Button>

              <Button variant={q1===false?"":"outline"} onClick={()=>setQ1(false)}>Yanlış</Button>

            </div>

          </Card>


          <Card>

            <SectionTitle>A2: Kaynağı Bul</SectionTitle>

            <div className="flex flex-wrap gap-2">

              {[news.source,"Kaynak A","Kaynak B"].map((o,i)=>(

                <Button key={i} variant={q2===i?"":"outline"} onClick={()=>setQ2(i)}>{o}</Button>

              ))}

            </div>

          </Card>


          <Card>

            <SectionTitle>A3: Kanıt ↔ Yorum</SectionTitle>

            <input type="range" min="0" max="100" value={q3} onChange={(e)=>setQ3(Number(e.target.value))} className="w-full"/>

            <div className="flex justify-between text-xs mt-1" style={{color:COLORS.dim}}>

              <span>Kanıt</span><span>Yorum</span>

            </div>

          </Card>


          <Card>

            <SectionTitle>A4: 7 Günlük Tahmin</SectionTitle>

            <div className="flex flex-wrap gap-2">

              {["Onay süreci hızlanır","Ek düzenleme ile uzar","Süreçte rafa kalkar"].map((o,i)=>(

                <Button key={i} variant={q4===i?"":"outline"} onClick={()=>setQ4(i)}>{o}</Button>

              ))}

            </div>

            <Text dim>Not: Puanın 7 gün sonra netleşir (şimdilik provizyonel).</Text>

          </Card>


          <Card>

            <SectionTitle>A5: Meme Başlığı</SectionTitle>

            <input value={q5} onChange={(e)=>setQ5(e.target.value)}

                   className="w-full bg-transparent rounded-xl p-3 border"

                   style={{borderColor:COLORS.border, color:COLORS.text}}

                   placeholder="Komik/yaratıcı bir başlık yaz"/>

          </Card>


          <div className="flex gap-2">

            <Button onClick={()=>setStep("killcard")}>Skoru Gör</Button>

            <Button variant="outline" onClick={()=>setStep("summary")}>Geri</Button>

          </div>

        </div>

      );


      const KillCard=()=>{

        const svgRef=useRef(null);

        useEffect(()=>{ /* skor Summary→Quiz→KillCard geçişinde hesaplandı */ },[]);

        return (

          <div className="max-w-4xl mx-auto p-6 space-y-4">

            <Card>

              <SectionTitle>Kill Card (Paylaş)</SectionTitle>

              <div className="flex flex-col md:flex-row gap-6 items-start">

                <div className="w-full md:w-1/2">

                  <KillCardSVG title={news.title} source={news.source} score={score} streak={streakMult} bullets={bullets} svgRef={svgRef}/>

                </div>

                <div className="w-full md:w-1/2 space-y-3">

                  <Text>Skor; doğru/yanlış, kaynak isabeti, slider, provizyonel tahmin ve meme başlığından oluşur. Streak çarpanı eklenir.</Text>

                  <div className="flex gap-2">

                    <Button onClick={()=>downloadKillCardPNG(svgRef,"killcard.png")}>PNG indir</Button>

                    <Button variant="outline" onClick={()=>setStep("landing")}>Ana sayfa</Button>

                  </div>

                  <Card>

                    <SectionTitle>Özet</SectionTitle>

                    <ul className="list-disc pl-6" style={{color:COLORS.text}}>

                      {bullets.map((b,i)=>(<li key={i}>{b}</li>))}

                    </ul>

                  </Card>

                </div>

              </div>

            </Card>

          </div>

        );

      };


      return (

        <div style={{background:COLORS.bg, minHeight:"100vh", fontFamily:"Inter, system-ui"}}>

          {step==="landing" && <Landing/>}

          {step==="summary" && <Summary/>}

          {step==="quiz" && <Quiz/>}

          {step==="killcard" && <KillCard/>}

          <footer className="text-center text-xs py-6" style={{color:COLORS.dim}}>

            © BossFight Gündem – Günlük limit: 3 haber • Bu bir demo arayüzüdür.

          </footer>

        </div>

      );

    }
